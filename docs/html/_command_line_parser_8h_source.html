102 template<typename T>
103 auto CommandLineParser::assertValueIn(
104 const Option<T> & opt,
105 const std::set<std::string>& allowed ) -> void
106 {
107 if( opt.isSet() && allowed.find( opt.get()) == allowed.end())
108 {
109 std::ostringstream oss;
110
111 oss << "Invalid value for " << opt.getLongFlag() << ": '" << opt.get() << "'. Allowed values are: ";
112
113 int count = 0;
114 for( const auto& val : allowed)
115 {
116 oss << (count++ != 0 ? "," : "") << "'" << val << "'";
117 }
118
119 throw std::runtime_error( oss.str());
120 }
