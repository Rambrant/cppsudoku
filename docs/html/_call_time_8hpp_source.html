21 template <typename Func, typename... Args>
22 auto CallTime( Func&& func, Args&&... args)
23 {
24 using ResultType = std::invoke_result_t< Func, Args...>;
25 using DurationResolution = std::chrono::microseconds;
26
27 const auto start = std::chrono::high_resolution_clock::now();
28
29 if constexpr( std::is_void_v< ResultType>)
30 {
31 std::invoke( std::forward<Func>( func), std::forward<Args>( args)...);
32
33 const auto end = std::chrono::high_resolution_clock::now();
34 const auto duration = std::chrono::duration_cast< DurationResolution>( end - start);
35
36 return std::make_tuple( std::monostate{}, duration);
37 }
38 else
39 {
40 const auto result = std::invoke( std::forward<Func>( func), std::forward<Args>( args)...);
41
42 const auto end = std::chrono::high_resolution_clock::now();
43 const auto duration = std::chrono::duration_cast< DurationResolution>( end - start);
44
45 return std::make_tuple( result, duration);
46 }
