33 std::ostringstream os;

172template <typename Func, typename... Args>
173auto timedCall(Func&& func, Args&&... args)
174{
175 auto start = std::chrono::high_resolution_clock::now();
176
177 auto result = std::invoke( std::forward<Func>( func), std::forward<Args>( args)...);
