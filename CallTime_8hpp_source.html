23 template <typename Func, typename... Args>
24 auto CallTime( Func&& func, Args&&... args)
25 {
26 using ResultType = std::invoke_result_t< Func, Args...>;
27 using DurationResolution = std::chrono::microseconds;
28
29 const auto start = std::chrono::high_resolution_clock::now();
30
31 if constexpr( std::is_void_v< ResultType>)
32 {
33 std::invoke( std::forward<Func>( func), std::forward<Args>( args)...);
34
35 const auto end = std::chrono::high_resolution_clock::now();
36 const auto duration = std::chrono::duration_cast< DurationResolution>( end - start);
37
38 return std::make_tuple( std::monostate{}, duration);
39 }
40 else
41 {
42 const auto result = std::invoke( std::forward<Func>( func), std::forward<Args>( args)...);
43
44 const auto end = std::chrono::high_resolution_clock::now();
45 const auto duration = std::chrono::duration_cast< DurationResolution>( end - start);
46
47 return std::make_tuple( result, duration);
